62      /**
63       * Verify the method returns true when there's a simple duplication (meaning, the names match).
64       * This check is only concerned if we found a duplicate. It should still fail even if the two *.class
65       * files are exactly the same.
66       */
67      @Test
68      public void hasDuplicatesShouldReturnTrueWhenClassNameIsDuplicate() throws Exception
69      {
70          ClassFile classFile1 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "" );
71          ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "" );
72          ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2 );
73  
74          boolean result = classesWithSameName.hasDuplicates( DETERMINE_DUPLICATES_BY_NAME );
75  
76          assertTrue( result );
77      }

81      {

83          ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "content matches in both" );
84          ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2 );

86          boolean result = classesWithSameName.hasDuplicates( DETERMINE_DUPLICATES_BY_NAME_AND_BYTECODE );

102     /**
103      * This test compares two files with the same exact relative path (so they look like the same file)
104      * but they exist in two different folders and their bytecode doesn't match. This should be considered
105      * a duplicate.
106      *
107      * We set the test up so it fails if it finds the same class name/path twice (meaning, it does not compare
108      * bytecode).
109      */
110     @Test
111     public void hasDuplicatesShouldReturnTrueWhenClassNameIsDuplicateButBytecodeDiffers() throws Exception
112     {
113         ClassFile classFile1 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "1" );
114         ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "2" );
115         ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2 );
116 
117         boolean result = classesWithSameName.hasDuplicates( DETERMINE_DUPLICATES_BY_NAME );
118 
119         assertTrue( result );
120     }

156     /**
157      * Verify the output string contains all the information, specifically, it should list which artifacts
158      * were an exact match (meaning, the bytecode of the .class files were identical). This helps users
159      * determine which artifacts they can ignore when fix the BanDuplicateClasses error.
160      */
161     @Test
162     public void toOutputStringOutputsTwoArtifactsWhereBytecodeIsExactMatch() throws Exception
163     {
164         ClassFile classFile1 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "content matches in both" );
165         ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "content matches in both" );
166         ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2 );
167 
168         String actualOutput = classesWithSameName.toOutputString( DETERMINE_DUPLICATES_BY_NAME_AND_BYTECODE );
169 
170         String expectedOutput = PATH_TO_CLASS_FILE + "  -- the bytecode exactly matches in these: " +
171             classFile1.getArtifactThisClassWasFoundIn() + " and " + classFile2.getArtifactThisClassWasFoundIn();
172         assertEquals( expectedOutput, actualOutput );
173     }
174 
175     /**
176      * This verifies the output string contains all the information, specifically, it should list
177      * which artifacts were an exact match. In this case we have 4 artifacts: 1, 2, 3, and 4.
178      * The bytecode of 1 and 2 match each other, the bytecode of 3 and 4 match each other, but
179      * 1 and 2 don't match 3 and 4.
180      */
181     @Test
182     public void toOutputStringOutputsFourArtifactsWhereBytecodeIsExactMatchInTwoAndExactMatchInOtherTwo() throws Exception
183     {
184         ClassFile classFile1 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "file content of 1 and 2" );
185         ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "file content of 1 and 2" );
186         ClassFile classFile3 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "file content of 3 and 4" );
187         ClassFile classFile4 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "file content of 3 and 4" );
188         ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2, classFile3, classFile4 );
189 
190         String actualOutput = classesWithSameName.toOutputString( DETERMINE_DUPLICATES_BY_NAME_AND_BYTECODE );
191 
192         String expectedOutput = PATH_TO_CLASS_FILE + "  -- the bytecode exactly matches in these: " +
193             classFile1.getArtifactThisClassWasFoundIn() + " and " + classFile2.getArtifactThisClassWasFoundIn() +
194             "; and more exact matches in these: " +
195             classFile3.getArtifactThisClassWasFoundIn() + " and " + classFile4.getArtifactThisClassWasFoundIn();
196         assertEquals( expectedOutput, actualOutput );
197     }

255     @Test
256     public void getAllArtifactsThisClassWasFoundInShouldReturnAllArtifacts() throws Exception
257     {
258         ClassFile classFile1 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "" );
259         ClassFile classFile2 = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "" );
260         ClassesWithSameName classesWithSameName = new ClassesWithSameName( log, classFile1, classFile2 );
261         Artifact artifact1 = classFile1.getArtifactThisClassWasFoundIn();
262         Artifact artifact2 = classFile2.getArtifactThisClassWasFoundIn();
263 
264         Set<Artifact> result = classesWithSameName.getAllArtifactsThisClassWasFoundIn();
265 
266         assertEquals( 2, result.size() );
267         assertTrue( result.contains( artifact1 ) );
268         assertTrue( result.contains( artifact2 ) );
