3   import org.junit.Rule;
4   import org.junit.Test;
5   import org.junit.rules.TemporaryFolder;
6   
7   import static org.junit.Assert.assertEquals;
8   
9   public class ClassFileTest
10  {
11      private static final String PATH_TO_CLASS_FILE = ClassFileTest.class.getName().replace( '.', '/' ) + ".class";
12  
13      @Rule
14      public TemporaryFolder tempFolder = new TemporaryFolder();
15  
16      private final ClassFileHelper classFileHelper = new ClassFileHelper( tempFolder );
17  
18      @Test
19      public void getHashComputesHashOfFile() throws Exception
20      {
21          ClassFile classFile = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "the content of the file" );
22  
23          assertEquals( "7e47820975c51a762e63caa95cc76e45", classFile.getHash() );
24      }
25  
26      @Test
27      public void getHashReturnsConsistentHashWhenInvokedTwice() throws Exception
28      {
29          ClassFile classFile = classFileHelper.createWithContent( PATH_TO_CLASS_FILE, "file content" );
30  
31          String hash1 = classFile.getHash();
32          String hash2 = classFile.getHash();
33  
34          assertEquals( "d10b4c3ff123b26dc068d43a8bef2d23", hash1 );
35          assertEquals( hash1, hash2 );
36      }
37  
38  }
